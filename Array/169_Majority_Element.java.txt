class Solution {
    public int majorityElement(int[] nums) {
        int n = nums.length;

        int count  = 0;
        int answer = nums[0];

        for(int i = 0; i < n; i++){
            if(nums[i] == answer){
                count++;

            } else if(count == 0){
                answer = nums[i];
                count = 1;

            }else {
                count--;
            }
        }


        return answer;

    }
}
