class Solution {
    public int missingNumber(int[] nums) {
        int n = nums.length;

        int lengthSum = (n * (n + 1) / 2);  // formula

        int sum = 0;

        for(int i = 0; i < n; i++){
            sum = sum + nums[i];
        }


        int missing = lengthSum - sum;

        return Math.abs(missing);
    }
}
